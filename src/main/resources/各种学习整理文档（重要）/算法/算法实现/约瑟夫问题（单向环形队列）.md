package com.algorithm;

public class Josepfu {
    public static void main(String[] args) {

        CircleLinkedList list = new CircleLinkedList();
        list.countBoy(3,5,20);
        //list.showBoys();
    }


}

class CircleLinkedList {
    private Boy first = null;

    public void addBoys(int nums) {
        if (nums < 0) {
            System.out.println("最少输入一个：");
        }

        Boy helper = first;
        for (int i = 0; i < nums; i++) {
            Boy boy = new Boy(i);

            if (i == 0) {
                first = boy;
                helper = first;
            } else {
                helper.setNext(boy);
                helper.getNext().setNext(first);
                helper = helper.getNext();
            }
        }
    }


    /**
     *
     * @param startNo 从哪个小孩开始数
     * @param countNum 总共数几次
     * @param circleSize 生产多大的圈
     */
    public void countBoy(int startNo,int countNum,int circleSize){
        if (startNo<1 || countNum<1 ||circleSize<1){
            System.out.println("开始的小孩节点或者数的次数或者生产的圈的大小不能小于1");
        }

        if (countNum>=circleSize){
            System.out.printf("大小为%d的圈，数的次数不能大于%d次",circleSize,circleSize-1);
        }
        //生产圈
        addBoys(circleSize);
        //开始数  这个时候first节点没必要一直指向第一个节点了
        Boy helper = first;

        //让helper指向first后一个节点
        while (true){
            if (helper.getNext()==first) break;
            helper=helper.getNext();
        }

        //让first指向开始数数节点和helper指向后一节点
        if (startNo!=1){
            for (int i = 0; i < startNo-1; i++) {
                first=first.getNext();
                helper=helper.getNext();
            }
        }

        while (true){
            if (first==helper) break;

            for (int i = 0; i < countNum-1; i++) {
                first=first.getNext();
                helper=helper.getNext();
            }
            System.out.printf("小孩编号%d出圈",first.getNo());
            System.out.println("");
            first=first.getNext();
            helper.setNext(first);
        }

    }

    public void showBoys() {
        if (first == null) {
            System.out.println("没有节点可展示！");
        }

        Boy helper = first;

        while (true) {
            System.out.printf("节点编号%d", helper.getNo());
            System.out.println("");
            if (helper.getNext() == first) break;
            helper = helper.getNext();
        }
    }

}


class Boy {
    private int no;//编号

    private Boy next;//下一个节点位置

    public Boy(int no) {
        this.no = no;
    }


    public int getNo() {
        return no;
    }

    public void setNo(int no) {
        this.no = no;
    }

    public Boy getNext() {
        return next;
    }

    public void setNext(Boy next) {
        this.next = next;
    }
}
